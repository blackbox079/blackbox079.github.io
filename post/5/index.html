Vue I18n 是 Vue.js 的国际化插件。本文搭配 iView 使用。

import ViewUI from 'view-design';
import VueI18n from 'vue-i18n';
import enView from 'view-design/dist/locale/en-US';
import zhView from 'view-design/dist/locale/zh-CN';
import en from './locales/en.json';
import zh from './locales/zh.json';
import 'view-design/dist/styles/iview.css';

Vue.use(ViewUI);
Vue.use(VueI18n);
Vue.locale = () => {};

// 合并本地语言包和iView框架语言包
const messages = {
    en: Object.assign(en, enView),
    zh: Object.assign(zh, zhView),
};
const i18n = new VueI18n({
    locale: localStorage.getItem('language') || 'zh',
    messages,
});

new Vue({
    router,
    i18n,
    store,
    render: (h) => h(App),
}).$mount('#app');

    <button class="change__button menu__item">
        {{ $t('language.change') }}
        <span class="t"></span>
    </button>
    <DropdownMenu slot="list">
        <DropdownItem v-for="(item, index) in languages" :key="index" :name="item.value">

        </DropdownItem>
    </DropdownMenu>
</Dropdown>
//移动端
<div class="language">
    <a @click="changeLanguage('zh')">简体</a>
    <div class="line"></div>
    <a @click="changeLanguage('en')">English</a>

export default {
    //...
    data() {
        return {
            locale: 'zh',
            languages: [
                { value: 'zh', label: '简体中文' },
                { value: 'en', label: 'English' },
            ],
        };
    },
    methods: {
        changeLanguage(value) {
            this.$i18n.locale = this.locale = value;
            localStorage.setItem('language', this.locale);
        },
    },
};
